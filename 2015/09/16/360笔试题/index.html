public class Test362 {

//判断素数
public static boolean issushu(int shuzi) {
    if (shuzi==2) {
        return true;
    }
    int max = (int)Math.sqrt(shuzi);
    for(int i=2;i<=max;i++)
    {    
        if (shuzi%i==0) {
            return false;
        }
    }        
    return true;            
}

//拆分因子，拼接成字符串
public static String  getshuzu(int value) { 
    String string = new String("");
    if (issushu(value)) {
        return string+"*"+value;
    }
    else{
        for(int i=2;i<=value/2;i++)
        {    
            if (value%i==0) {
                //if(issushu(i))
                    string = string+"*"+i;
                //else
                //    string = string+getshuzu(i);
                if(issushu(value/i))
                    string = string+"*"+value/i;
                else
                    string = string+getshuzu(value/i);    

                break;
            }                            
        }
        return string;
    }        
}

public static List<Integer> primeFactor(int n) {// 该方法将一个正整数分解成质因数相乘的形式  

    List<Integer> resultlist = new LinkedList<Integer>();
            int num = n; 
            if (num < 2) {// 若输入的数小于2,输出提示信息 
                resultlist.add(num);
                return resultlist;
            } 
            else{ 
                int primeNumber = 2;// 定义最小的质数 
                while (primeNumber <= num) {// 在质数小于输入的数时，进行循环 
                    if (primeNumber == num) {// 当质数等于输入的数时,直接输出 
                        resultlist.add(num);
                        break;// 跳出循环
                    } 
                    else if (num % primeNumber == 0) {// 当输入的数与质数的余数为0时,输出这个质数
                        resultlist.add(primeNumber);
                        num = num / primeNumber;// 把剩下的结果赋给num

                    }
                    else {// 在余数不为0时,质数递增
                        primeNumber++;
                    }
                }
            }
            return resultlist; 
}
